# ADR 009: RPC Contract

## Status

Approved

## Context

Internal service communication uses gRPC. The contracts between Gateway ↔ Room Service ↔ Game Server must be defined.

## Problem

Internal service communication must use consistent, well-defined contracts. Ad-hoc APIs lead to tight coupling, versioning issues, and integration bugs as the system grows.

## Decision

### Service Definitions (protobuf)

```protobuf
// === spatial_server_api.proto (Business Backend → Spatial Server) ===

service SpatialServerAPI {
  // Business Backend creates a realtime runtime
  rpc CreateRuntime(CreateRuntimeRequest) returns (CreateRuntimeResponse);
  // Business Backend destroys a runtime and releases all resources
  rpc DestroyRuntime(DestroyRuntimeRequest) returns (DestroyRuntimeResponse);
  // Business Backend queries runtime status
  rpc GetRuntimeInfo(GetRuntimeInfoRequest) returns (GetRuntimeInfoResponse);
  // Business Backend queries runtime metrics
  rpc GetRuntimeMetrics(GetRuntimeMetricsRequest) returns (GetRuntimeMetricsResponse);
  // Business Backend lists all runtimes
  rpc ListRuntimes(ListRuntimesRequest) returns (ListRuntimesResponse);
}

message CreateRuntimeRequest {
  string runtime_id = 1;       // UUIDv7, generated by Business Backend
  int32 zone_count = 2;        // initial number of zones
  double zone_size = 3;        // configurable zone size (optional)
  string game_server_type = 4; // optional: pin to specific Game Server type
}

message CreateRuntimeResponse {
  string runtime_id = 1;
  string gateway_addr = 2;     // connection address for clients
  int32 zone_count = 3;
  repeated ZoneInfo zones = 4;
}

message ZoneInfo {
  string zone_id = 1;
  int32 grid_x = 2;
  int32 grid_y = 3;
  string game_server_id = 4;
}

message DestroyRuntimeRequest {
  string runtime_id = 1;
}

message DestroyRuntimeResponse {
  bool success = 1;
}

message GetRuntimeInfoRequest {
  string runtime_id = 1;
}

message GetRuntimeInfoResponse {
  string runtime_id = 1;
  RuntimeStatus status = 2;    // enum: CREATING, ACTIVE, DRAINING, DESTROYED
  int32 zone_count = 3;
  int32 player_count = 4;
  string created_at = 5;
}

message GetRuntimeMetricsRequest {
  string runtime_id = 1;
}

message GetRuntimeMetricsResponse {
  string runtime_id = 1;
  int32 player_count = 2;
  double avg_latency_ms = 3;
  double cpu_usage = 4;
  double memory_usage = 5;
}

message ListRuntimesRequest {
  int32 page_size = 1;
  string page_token = 2;
}

message ListRuntimesResponse {
  repeated GetRuntimeInfoResponse runtimes = 1;
  string next_page_token = 2;
}

// === common.proto ===

message EntityID {
  string id = 1;  // UUIDv7
}

message ZoneID {
  string id = 1;           // zone identifier
  string runtime_id = 2;   // runtime the zone belongs to
  int32 grid_x = 3;
  int32 grid_y = 4;
}

message RuntimeID {
  string id = 1;
}

message ServerID {
  string id = 1;
}

message Vector3 {
  double x = 1;
  double y = 2;
  double z = 3;
}

// Enums shared across services (see common.proto for full definitions)
enum RuntimeStatus {
  RUNTIME_STATUS_UNSPECIFIED = 0;
  RUNTIME_STATUS_CREATING = 1;
  RUNTIME_STATUS_ACTIVE = 2;
  RUNTIME_STATUS_DRAINING = 3;
  RUNTIME_STATUS_DESTROYED = 4;
}

enum ZoneStatus {
  ZONE_STATUS_UNSPECIFIED = 0;
  ZONE_STATUS_UNOWNED = 1;
  ZONE_STATUS_ACTIVE = 2;
  ZONE_STATUS_TRANSFERRING = 3;
  ZONE_STATUS_ORPHAN = 4;
}

enum ServerStatus {
  SERVER_STATUS_UNSPECIFIED = 0;
  SERVER_STATUS_JOINING = 1;
  SERVER_STATUS_ACTIVE = 2;
  SERVER_STATUS_DRAINING = 3;
  SERVER_STATUS_SHUTDOWN = 4;
}

// === room_service.proto ===

service RoomService {
  // Game Server → Room Service
  rpc Register(RegisterRequest) returns (RegisterResponse);
  rpc Heartbeat(HeartbeatRequest) returns (HeartbeatResponse);
  rpc PrepareShutdown(ServerID) returns (PrepareShutdownResponse);

  // Gateway → Room Service
  rpc LookupZone(LookupZoneRequest) returns (LookupZoneResponse);
  rpc LookupServer(LookupServerRequest) returns (LookupServerResponse);
  rpc ReportMetrics(ReportMetricsRequest) returns (ReportMetricsResponse);

  // Room Service → Game Server (control plane)
  rpc TransferZone(stream ZoneSnapshot) returns (TransferZoneResponse);
  rpc PrepareTransfer(PrepareTransferRequest) returns (PrepareTransferResponse);
}

// NOTE: TransferZone is a CONTROL-PLANE RPC. Room Service uses it to instruct a
// server to accept ownership of a zone. The actual zone state (entities, AOI
// index) flows via direct GameServer.ZoneStateSync (P2P, source → target),
// NOT through RoomService.TransferZone. See ADR-002 for the data-plane flow.


message RegisterRequest {
  string server_id = 1;
  string host = 2;
  int32 port = 3;
  int32 max_zones = 4;
  map<string, string> metadata = 5;
}

message RegisterResponse {
  bool success = 1;
}

message HeartbeatRequest {
  string server_id = 1;
  // NOTE: HeartbeatRequest has NO load field. It is a liveness signal only.
  // Load reporting will be via the ReportMetrics RPC (not yet implemented).
}

message HeartbeatResponse {
  bool acknowledged = 1;
}

message LookupZoneRequest {
  string zone_id = 1;
}

message LookupZoneResponse {
  ServerID server = 1;
  string host = 2;
  int32 port = 3;
}

message LookupServerRequest {
  string server_id = 1;
}

message LookupServerResponse {
  string host = 1;
  int32 port = 2;
  ServerStatus status = 3;
}

message ReportMetricsRequest {
  double cpu_usage = 1;
  double memory_usage = 2;
  int32 active_connections = 3;
  int32 active_runtimes = 4;
}

message ReportMetricsResponse {
  bool acknowledged = 1;
}

// === game_server.proto ===

service GameServer {
  // Room Service → Game Server (runtime lifecycle)
  rpc AssignZone(AssignZoneRequest) returns (AssignZoneResponse);
  rpc ReleaseZone(ReleaseZoneRequest) returns (ReleaseZoneResponse);

  // Game Server → Game Server (direct)
  rpc MigrateEntity(MigrateEntityRequest) returns (MigrateEntityResponse);
  rpc ZoneStateSync(stream ZoneSnapshot) returns (ZoneStateSyncResponse);
  rpc NotifyEntityEnter(EntityEnterLeave) returns (NotifyResponse);
  rpc NotifyEntityLeave(EntityEnterLeave) returns (NotifyResponse);
  rpc SendEntityUpdate(EntityUpdate) returns (Ack);
  rpc QueryEntities(QueryEntitiesRequest) returns (QueryEntitiesResponse);

  // Gateway ↔ Game Server (client packet relay)
  rpc Relay(stream RelayPacket) returns (stream RelayPacket);
}

// Relay envelope carried over the Gateway ↔ Game Server bidirectional stream.
enum Kind {
  KIND_UNSPECIFIED = 0;
  KIND_DATA = 1;
  KIND_CONNECT = 2;
  KIND_DISCONNECT = 3;
}

message RelayPacket {
  string      client_id = 1;
  Kind        kind      = 2;
  bytes       payload   = 3;
  ConnectMeta meta      = 4;
}

message ConnectMeta {
  string player_id  = 1;
  string runtime_id = 2;
  string zone_id    = 3;
}

message MigrateEntityRequest {
  EntitySnapshot entity = 1;
  string target_zone_id = 2;
}

message MigrateEntityResponse {
  bool success = 1;
}

message ZoneStateSyncResponse {
  bool success = 1;
}

message NotifyResponse {
  bool acknowledged = 1;
}

message EntitySnapshot {
  string entity_id = 1;
  string type = 2;
  Vector3 position = 3;
  map<string, bytes> attributes = 4;
}

message ZoneSnapshot {
  ZoneID zone_id = 1;
  repeated EntitySnapshot entities = 2;
  bytes aoi_state = 3;  // serialized in-memory AOI index
}

message AssignZoneRequest {
  string zone_id = 1;
  string runtime_id = 2;
  int32 grid_x = 3;
  int32 grid_y = 4;
  double zone_size = 5;
}

message AssignZoneResponse {
  bool success = 1;
}

message ReleaseZoneRequest {
  string zone_id = 1;
  string runtime_id = 2;
}

message ReleaseZoneResponse {
  bool success = 1;
}

message EntityUpdate {
  string entity_id = 1;
  Vector3 position = 2;
  int32 sequence = 3;
  int64 timestamp = 4;
}

message QueryEntitiesRequest {
  int32 grid_x = 1;
  int32 grid_y = 2;
  double radius = 3;
  repeated string type_filter = 4;
}

message QueryEntitiesResponse {
  repeated EntitySnapshot entities = 1;
}

message Ack {
  bool success = 1;
}

// === gateway.proto ===

service Gateway {
  // Client-facing messages are NOT defined here (they use the binary packet protocol)
  // This is for inter-service gRPC only
}
```

> **Note:** Per [ADR-002](002-zone-migration.md), zone state flows via direct `GameServer.ZoneStateSync` (P2P, source → target), NOT through `RoomService.TransferZone`. `TransferZone` is a control-plane instruction only — Room Service directs a transfer, while the actual entity/AOI data is streamed directly between Game Servers.

> **Note:** `HeartbeatRequest` carries only `server_id` (liveness signal). It has no load field. Load reporting will be via the `ReportMetrics` RPC (not yet implemented).

| Register | 5s | 3x | Yes | No |
| Heartbeat | 2s | No | Yes | No |
| LookupZone | 2s | 3x | Yes | No |
| ReportMetrics | 2s | 2x (best effort) | Yes | No |
| AssignZone | 5s | 2x | Yes | No |
| ReleaseZone | 5s | 2x | Yes | No |
| TransferZone | 30s (stream) | 1x | Yes | gzip |
| PrepareTransfer | 5s | 1x | Yes | No |
| MigrateEntity | 5s | 2x | Yes | No |
| ZoneStateSync | 30s (stream) | 1x | Yes | gzip |
| PrepareShutdown | 5s | 1x | Yes | No |
| SendEntityUpdate | 500ms | No (latest wins) | No | No |
| QueryEntities | 1s | 1x | Yes | No |
| Relay | stream (no timeout) | No | No | No |
| CreateRuntime | 10s | 2x | Yes | No |
| DestroyRuntime | 30s | 2x | Yes | No |
| GetRuntimeInfo | 5s | 2x | Yes | No |
| GetRuntimeMetrics | 5s | 2x | Yes | No |
| ListRuntimes | 10s | 2x | Yes | No |

### Error Codes

| Code | Meaning |
|------|---------|
| `ZONE_NOT_FOUND` | Zone ID does not exist |
| `ZONE_NOT_OWNED` | Server does not own this zone |
| `SERVER_NOT_FOUND` | Server ID not registered |
| `TRANSFER_IN_PROGRESS` | Zone is already being transferred |
| `SERVER_DRAINING` | Server is in DRAINING state |
| `RATE_LIMITED` | Client is rate limited |
| `RUNTIME_NOT_FOUND` | Runtime ID does not exist |
| `RUNTIME_ALREADY_EXISTS` | Runtime ID already exists |
| `NO_SERVER_AVAILABLE` | No Game Server available to host the runtime |

## Alternatives

1. **REST/JSON APIs**: Simpler to debug and test with standard HTTP tooling. Higher bandwidth overhead, no built-in streaming support, and no code generation from schema.
2. **Message queue (NATS/RabbitMQ)**: Async communication decouples senders and receivers. Higher latency for request-response patterns and more infrastructure to manage.
3. **Raw TCP with custom framing**: Maximum control over wire format. No code generation, no schema enforcement, and poor ecosystem tooling.

## Tradeoffs

- gRPC provides strong typing, streaming, and code generation but adds protobuf compilation to the build pipeline.
- Streaming RPCs enable efficient zone state transfer but require persistent connection management.
- Error codes in protobuf are well-typed but require client libraries to handle them consistently.

## Consequences

- Proto3 definitions are the source of truth — all code generation from protos.
- gzip compression on streaming RPCs (zone transfer) but not on hot-path RPCs.
- UUIDv7 for all entity and zone IDs.
- SpatialServerAPI is exposed to Business Backend via gRPC (or REST gateway later).
- Gateway address is returned to Business Backend so it can relay to client.
- Runtime lifecycle is fully managed by Business Backend (create → use → destroy).
- Spatial Server never stores business metadata (only runtime operational state).

## Future Considerations

- gRPC-Web for browser-based Game Server admin tools.
- Schema registry for protobuf version management.
- Migration to gRPC bidirectional streaming for realtime entity sync.
